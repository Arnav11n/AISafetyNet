Step 3: Create the Talking Video (MAIN PART)
Example using JavaScript (Node.js)
import fetch from "node-fetch";

const HEYGEN_API_KEY = "YOUR_API_KEY_HERE";

const response = await fetch("https://api.heygen.com/v1/video.generate", {
  method: "POST",
  headers: {
    "Authorization": `Bearer ${HEYGEN_API_KEY}`,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    video_inputs: [
      {
        character: {
          type: "photo",
          photo_url: "https://yourstorage.com/user_photo.jpg"
        },
        voice: {
          type: "text",
          input_text: "Hello, this is a demo video generated using AI."
        }
      }
    ],
    dimension: {
      width: 720,
      height: 1280
    }
  })
});

const data = await response.json();
console.log(data);
